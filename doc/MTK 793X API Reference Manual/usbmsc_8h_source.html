45 /* Possible values for quirks field. */
46 enum {
47  /* Don't check for LUNs (force assumption that there's only one LUN). */
48  USB_MSC_QUIRK_NO_LUNS = 1 << 0,
49  /* Never do a BULK_ONLY reset, just continue. This means that the device
50  cannot recover from phase errors and won't detach automatically for
51  unrecoverable errors. Do not use unless you have to. */
52  USB_MSC_QUIRK_NO_RESET = 1 << 1,
53 };

57  USB_MSC_DETACHED = -1, /* Disk detached or out to lunch. */
58  USB_MSC_NOT_READY = 0, /* Disk not ready yet -- empty card reader */
59  USB_MSC_READY = 1, /* Disk ready to communicate. */
60 };
