126 struct uac1_header_descriptor {
127  u8 bLength; /* 8 + n */
128  u8 bDescriptorType; /* USB_DT_CS_INTERFACE */
129  u8 bDescriptorSubtype; /* UAC_MS_HEADER */
130  u16 bcdADC; /* 0x0100 */

134 } __attribute__((packed));

143  u8 bAssocTerminal; /* ID of the Output Terminal associated */
